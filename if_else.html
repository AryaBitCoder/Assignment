4. Find the output of the following code snippets.

1 
int x = 10, y = 20;
if (x > y) {
    printf("A");
} else if (x < y) {
    printf("B");
} else {
    printf("C");
}
2 
char grade = 'B';
switch (grade)
{
    case 'A':
        printf("Excellent");
        break;
    case 'B':
        printf("Good");
        break;
    case 'C':
        printf("Average");
        break;
    default:
        printf("Needs Improvement");
}
3 
int a = 10;
float b = 2.0;
switch(a / b)
{
    case 0:
        printf("Zero");
        break;
    default:
        printf("Non-zero");
}
4 
int a = 1;
switch(a)
{
    case 0:
        printf("Zero");
    case 1:
        printf("One");
    case 2:
        printf("Two");
        break;
    default:
        printf("Non-zero");
}
5 
switch((10 > 2) + (10 < 2) + (10 >= 2))
{
    case 0:
        printf("Zero");
        break;
    case 1:
        printf("One");
        break;
    case 2:
        printf("Two");
        break;
    default:
        printf("Other");
}
6 
int a = 15, b = 20;
int result = (a > b) ? (a - b) : (b - a);
printf("%d\n", result);
7 
int p = 8, q = 5, r = 12;
int max = (p > q) ? ((p>r) ? p : r) : ((q>r) ? q : r);
printf("%d\n", max);
